\subsection*{Problem 78}

Let $x_1,\dots,x_n$ have mean $\bar x$ and sample sd $s$.

\begin{enumerate}
\item[(a)] $y_i=x_i-\bar x$:
\[
\bar y=\frac{1}{n}\sum (x_i-\bar x)=\frac{1}{n}\left(\sum x_i-n\bar x\right)=0,
\]
\[
s_y^2=\frac{1}{n-1}\sum (y_i-\bar y)^2=\frac{1}{n-1}\sum (x_i-\bar x)^2=s^2.
\]

\item[(b)] $z_i=\dfrac{x_i-\bar x}{s}$:
\[
\bar z=\frac{1}{n}\sum \frac{x_i-\bar x}{s}=\frac{1}{s}\cdot \frac{1}{n}\sum (x_i-\bar x)=0,
\]
\[
s_z^2=\frac{1}{n-1}\sum (z_i-\bar z)^2
=\frac{1}{n-1}\sum \left(\frac{x_i-\bar x}{s}\right)^2
=\frac{1}{s^2}\cdot \frac{1}{n-1}\sum (x_i-\bar x)^2
=\frac{s^2}{s^2}=1.
\]
\end{enumerate}
